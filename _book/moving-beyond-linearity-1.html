Capítulo 5 Moving Beyond Linearity 

library(ISLR)
library(magrittr)

# Defino el tipo de modelo

parsnip::linear_reg() %>%
    parsnip::set_engine("lm") %>%
    parsnip::set_mode('regression') %>%
    assign(
        "lm_mod",
        .,
        envir = .GlobalEnv
    )

# Receta : usando step_bs 

recipes::recipe(
    wage ~ age,
    data = Wage
) %>%
    recipes::step_bs(
        age,
        options = list(
          knots = c(25,50,60)
        )
    ) %>%
    recipes::prep() %>%
    assign(
        "recipe_bs",
        .,
        envir = .GlobalEnv
    )

# Usando la receta (Si lo hago de esta manera me da diferente al libro!)

workflows::workflow() %>%
    workflows::add_model(lm_mod) %>%
    workflows::add_recipe(recipe_bs) %>%
    generics::fit(
        data = Wage
    ) %>%
    assign(
        "model_wkf",
        .,
        envir = .GlobalEnv
    )

broom::tidy(model_wkf) %>%
    gt::gt()
